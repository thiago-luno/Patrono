# Instalação Angular CLI
npm install -g @angular/cli@6.0.7

# Criação de novo Projeto
ng new nome_do_projeto

# Rodar projeto
ng serve --open

# Instalação rxjs versão que funciona
npm install rxjs@6.0.0 --save

# Instação Bootstrap
npm install bootstrap@4.1.1
